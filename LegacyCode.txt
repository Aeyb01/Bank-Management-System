        public static (int column, int row) FullName()
        {

                row++;

                if (line.Contains("Full Name:"))
                {

                    break;
                }
            }

            // We'll need to find the right column

            string text = lines[row];
            
            column = text.IndexOf(":");

            column += 5;

            return(column, row);
        }
